00037 inline Matrix26d
00038 frameJac(const SE3 & T,
00039          const Vector3d & xyz,
00040          const Vector2d & focal_length)
00041 {
00042   const Vector3d & xyz_trans = T*xyz;
00043   double x = xyz_trans[0];
00044   double y = xyz_trans[1];
00045   double z = xyz_trans[2];
00046   double z_2 = z*z;
00047   Matrix26d frame_jac;
00048   frame_jac(0,0) = -1./z *focal_length[0];
00049   frame_jac(0,1) = 0;
00050   frame_jac(0,2) = x/z_2 *focal_length[0];
00051   frame_jac(0,3) =  x*y/z_2 * focal_length[0];
00052   frame_jac(0,4) = -(1+(x*x/z_2)) *focal_length[0];
00053   frame_jac(0,5) = y/z *focal_length[0];
00054 
00055   frame_jac(1,0) = 0;
00056   frame_jac(1,1) = -1./z *focal_length[1];
00057   frame_jac(1,2) = y/z_2 *focal_length[1];
00058   frame_jac(1,3) = (1+y*y/z_2) *focal_length[1];
00059   frame_jac(1,4) = -x*y/z_2 *focal_length[1];
00060   frame_jac(1,5) = -x/z *focal_length[1];
00061   return frame_jac;
00062 }
